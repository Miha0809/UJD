	\textbf{Algorithm 1} FindDuplicate
	\hline
	\begin{algorithm}
		\KwIn{A sequence of integers (${a_{1}, a_{2}, ..., a_{n}}$)}
		\KwOut{\textsc{The index of first location with the same value as in a previous location in the sequence}}
		$location \gets 0$ i $\gets 2$ \\
		
		\While {$i \leq n$ $\wedge$  $location = 0$}
		{$j \gets 1$ \While{$j < i$ $\wedge$ $location = 0$}
			{\lIf{$a_{i}=a_{j}$}
				{$location \gets i$}
				\lElse{$j \gets j+1$}
			}
			$i \gets i+1$
		}
		\Return{location}
	\end{algorithm}
