	List-Search(L,k)
	\begin{algorithmic}[1]
		\STATE x:=head[L];
		\WHILE{(x!=NIL and key[x]!=k)}
			\STATE x:=next[x];
		\ENDWHILE
		\RETURN x;
	\end{algorithmic}
